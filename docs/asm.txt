mov <куда> <что> -> перемещает данные в регистр
example:
    mov rax 21 (поместить на rax 21)
    mov rdi rax (переместить значение из rax в rdi -> тоесть 21)


push <значение/регистр> -> добавляет значение в стек 
example:
    mov rax 1
    push rax (добавляет в стек значение из регистра rax -> тоесть 1)


pop <куда> -> удаляет из стека верхнее значение и перемещает в указанный регистр
example:
    mov rax 45
    push rax
    pop rdi (достает значение из стека и сохраняет его в регистр rdi)
